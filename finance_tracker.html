Шаг 1 — Создайте таблицу

Откройте sheets.google.com → создайте новую таблицу.


Шаг 2 — Apps Script

В таблице: Расширения → Apps Script → вставьте код ниже → сохраните (💾).


const SS = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { const action = e.parameter.action; if (action === 'get') return getData(); return ContentService.createTextOutput('ok'); } function doPost(e) { const data = JSON.parse(e.postData.contents); if (data.action === 'add') addRow(data); if (data.action === 'delete') deleteRow(data.id); if (data.action === 'addGoal') addGoal(data); return ContentService.createTextOutput('ok') .setMimeType(ContentService.MimeType.TEXT); } function getSheet(name) { return SS.getSheetByName(name) || SS.insertSheet(name); } function getData() { const txSheet = getSheet('Transactions'); const gSheet = getSheet('Goals'); const txRows = txSheet.getDataRange().getValues(); const gRows = gSheet.getDataRange().getValues(); const result = {transactions: [], goals: []}; for (let i = 1; i < txRows.length; i++) { const r = txRows[i]; result.transactions.push({id:r[0],type:r[1],amount:r[2],currency:r[3],category:r[4],note:r[5],date:r[6]}); } for (let i = 1; i < gRows.length; i++) { const r = gRows[i]; result.goals.push({id:r[0],name:r[1],goal:r[2]}); } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function addRow(data) { const sh = getSheet('Transactions'); if (sh.getLastRow() === 0) sh.appendRow(['id','type','amount','currency','category','note','date']); sh.appendRow([data.id,data.type,data.amount,data.currency,data.category,data.note,data.date]); } function deleteRow(id) { const sh = getSheet('Transactions'); const vals = sh.getDataRange().getValues(); for (let i = vals.length - 1; i >= 0; i--) { if (String(vals[i][0]) === String(id)) { sh.deleteRow(i + 1); break; } } } function addGoal(data) { const sh = getSheet('Goals'); if (sh.getLastRow() === 0) sh.appendRow(['id','name','goal']); sh.appendRow([data.id, data.name, data.goal]); }
